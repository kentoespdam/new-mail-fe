## Implementation Notes

### HelpFAB Component

```typescript
'use client';

import { useState } from 'react';
import { CircleHelp, MessageCircle, BookOpen, RefreshCw } from 'lucide-react';
import { useOnboarding } from '@/hooks/useOnboarding';
import { HelpCenterModal } from '@/components/help/HelpCenterModal';

export function HelpFAB() {
  const [isOpen, setIsOpen] = useState(false);
  const [isHelpCenterOpen, setIsHelpCenterOpen] = useState(false);
  const { resetTour } = useOnboarding();

  const handleRetour = async () => {
    await resetTour();
    setIsOpen(false);
    // Tour will re-trigger via useOnboarding state change
  };

  const handleContactSupport = () => {
    // Replace with actual IT support contact
    window.open('https://wa.me/6281234567890', '_blank');
    // Or: window.open('mailto:support@perumdamts.co.id');
    setIsOpen(false);
  };

  return (
    <>
      <div className="fixed bottom-6 right-6 z-50">
        <div className="relative">
          {isOpen && (
            <div className="absolute bottom-14 right-0 w-64 rounded-lg border bg-white shadow-lg">
              <div className="p-2 space-y-1">
                <button
                  onClick={handleRetour}
                  className="flex w-full items-center gap-2 rounded-md px-3 py-2 text-sm hover:bg-slate-100"
                >
                  <RefreshCw className="h-4 w-4" />
                  <span>Ulangi Panduan</span>
                </button>
                <button
                  onClick={handleContactSupport}
                  className="flex w-full items-center gap-2 rounded-md px-3 py-2 text-sm hover:bg-slate-100"
                >
                  <MessageCircle className="h-4 w-4" />
                  <span>Hubungi IT Support</span>
                </button>
                <button
                  onClick={() => {
                    setIsHelpCenterOpen(true);
                    setIsOpen(false);
                  }}
                  className="flex w-full items-center gap-2 rounded-md px-3 py-2 text-sm hover:bg-slate-100"
                >
                  <BookOpen className="h-4 w-4" />
                  <span>Pusat Bantuan</span>
                </button>
              </div>
            </div>
          )}
          
          <button
            onClick={() => setIsOpen(!isOpen)}
            className="flex h-12 w-12 items-center justify-center rounded-full bg-amber-500 text-white shadow-lg hover:bg-amber-600 transition-colors"
            aria-label="Bantuan"
          >
            <CircleHelp className="h-6 w-6" />
          </button>
        </div>
      </div>

      <HelpCenterModal
        isOpen={isHelpCenterOpen}
        onOpenChange={setIsHelpCenterOpen}
      />
    </>
  );
}
```
