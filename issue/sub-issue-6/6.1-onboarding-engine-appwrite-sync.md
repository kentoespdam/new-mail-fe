## Description

Membangun infrastruktur *tour guide* yang otomatis muncul bagi pengguna baru dan menyimpan status progresnya di Appwrite agar tidak mengganggu pengguna lama.

## Implementation Notes

### useOnboarding Hook

```typescript
import { useEffect, useState } from 'react';
import { account } from '@/lib/appwrite-rest';

interface OnboardingState {
  shouldRunTour: boolean;
  isLoading: boolean;
  markTourComplete: () => Promise<void>;
  resetTour: () => Promise<void>; // For "Ulangi Panduan"
}

export function useOnboarding(): OnboardingState {
  const [shouldRunTour, setShouldRunTour] = useState(false);
  const [isLoading, setIsLoading] = useState(true);

  useEffect(() => {
    const checkTourStatus = async () => {
      try {
        const prefs = await account.getPrefs();
        setShouldRunTour(!prefs.tour_completed);
      } catch (error) {
        console.error('Failed to get tour prefs:', error);
      } finally {
        setIsLoading(false);
      }
    };

    checkTourStatus();
  }, []);

  const markTourComplete = async () => {
    await account.updatePrefs({ tour_completed: true });
    setShouldRunTour(false);
  };

  const resetTour = async () => {
    await account.updatePrefs({ tour_completed: false });
    setShouldRunTour(true);
  };

  return { shouldRunTour, isLoading, markTourComplete, resetTour };
}
```

### React Joyride Integration

```typescript
import Joyride, { STATUS } from 'react-joyride';

const tourSteps = [
  {
    target: '.sidebar-tree',
    content: 'Navigasi folder surat Anda di sini',
    placement: 'right',
  },
  {
    target: '.toolbar-actions',
    content: 'Tombol aksi: Tulis Baru, Balas, Teruskan',
    placement: 'bottom',
  },
  // ... more steps
];

<Joyride
  steps={tourSteps}
  run={shouldRunTour}
  continuous
  showProgress
  showSkipButton
  skipButton={{ text: 'Lewati' }}
  callback={({ status }) => {
    if (status === STATUS.SKIPPED || status === STATUS.FINISHED) {
      markTourComplete();
    }
  }}
/>
```
