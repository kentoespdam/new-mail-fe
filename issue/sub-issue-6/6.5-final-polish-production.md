### JWT Cache Verification

```typescript
// lib/jwt-cache.ts - Verify cache handles concurrent requests
interface JWTCacheEntry {
  token: string;
  expiresAt: number;
  refreshPromise?: Promise<string>;
}

const jwtCache = new Map<string, JWTCacheEntry>();

export async function getValidJWT(sessionId: string): Promise<string> {
  const entry = jwtCache.get(sessionId);
  
  if (!entry || entry.expiresAt < Date.now()) {
    // Token expired or missing - refresh
    return refreshAndCache(sessionId);
  }
  
  if (entry.refreshPromise) {
    // Concurrent request - reuse same promise
    return entry.refreshPromise;
  }
  
  return entry.token;
}
```
