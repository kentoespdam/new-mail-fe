## Implementation Notes

### Layout Detection

```typescript
import { account } from '@/lib/appwrite-rest';

const getLayoutType = async (): Promise<'classic' | 'modern'> => {
  const prefs = await account.getPrefs();
  return prefs.theme_layout || 'modern'; // Default to modern
};
```

### Layout-Aware Tour Hook

```typescript
import { useState, useEffect } from 'react';
import { account } from '@/lib/appwrite-rest';

export function useLayoutAwareTour() {
  const [steps, setSteps] = useState<typeof classicSteps>([]);
  const [layout, setLayout] = useState<'classic' | 'modern'>('modern');

  useEffect(() => {
    const loadTourSteps = async () => {
      const userLayout = await getLayoutType();
      setLayout(userLayout);
      
      if (userLayout === 'classic') {
        setSteps(classicSteps);
      } else {
        setSteps(modernSteps);
      }
    };

    loadTourSteps();
  }, []);

  return { steps, layout };
}
```
