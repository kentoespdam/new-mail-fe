### Quick Messages Integration

```typescript
// Fetch quick messages
const { data: quickMessages } = useQuery({
  queryKey: ['quickMessages'],
  queryFn: () => quickMessagesService.getAll(),
});

// On selection
const handleQuickMessageSelect = (message: QuickMessage) => {
  setValue('content', message.template); // Auto-populate editor
  if (message.subject) {
    setValue('subject', message.subject);
  }
};
```

### Quick Message Interface

```typescript
interface QuickMessage {
  id: string;
  title: string;
  template: string;
  subject?: string;
  category?: string;
  isActive: boolean;
}
```
